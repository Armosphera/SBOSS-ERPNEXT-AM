import unittest
import frappe
from frappe.tests.utils import FrappeTestCase

class TestAMCompanySetup(FrappeTestCase):
    def test_doctype_exists(self):
        meta = frappe.get_meta("AM Company Setup")
        self.assertTrue(meta.custom)
        self.assertEqual(meta.module, "Armosphera Armenia Localization")

    def test_required_fields(self):
        meta = frappe.get_meta("AM Company Setup")
        field_names = {f.fieldname for f in meta.fields}
        for required in ("company", "taxpayer_identification_number",
                         "registration_number", "legal_form", "vat_treatment"):
            self.assertIn(required, field_names, f"missing field {required}")

    def test_tin_validation(self):
        """Armenian TIN (HHID / ՀՎՀՀ) is 8 digits, must validate."""
        from frappe_armenia.regional.doctype.am_company_setup.am_company_setup import validate_tin
        self.assertTrue(validate_tin("01234567"))
        self.assertFalse(validate_tin("1234567"))     # too short
        self.assertFalse(validate_tin("012345678"))   # too long
        self.assertFalse(validate_tin("abcdefgh"))    # not digits
        self.assertFalse(validate_tin(""))            # empty

    def test_single_doc_enforced(self):
        """Only one AM Company Setup row per company is allowed."""
        # After install, creating the first row is allowed.
        first = frappe.get_doc({
            "doctype": "AM Company Setup",
            "company": "_Test Company",
            "taxpayer_identification_number": "01234567",
            "registration_number": "12345678",
            "legal_form": "LLC",
            "vat_treatment": "Standard",
        }).insert(ignore_permissions=True)
        # Second insert for the same company should fail
        with self.assertRaises(frappe.exceptions.DuplicateEntryError):
            frappe.get_doc({
                "doctype": "AM Company Setup",
                "company": "_Test Company",
                "taxpayer_identification_number": "99999999",
                "registration_number": "88888888",
                "legal_form": "LLC",

            }).insert(ignore_permissions=True)

### Step 4 — Implement TIN validator
```python
# apps/frappe_armenia/frappe_armenia/regional/doctype/am_company_setup/am_company_setup.py
import frappe
from frappe.model.document import Document

class AMCompanySetup(Document):
    def validate(self):
        if not validate_tin(self.taxpayer_identification_number):
            frappe.throw(
                frappe._("TIN (HHID/ՀՎՀՀ) must be exactly 8 digits"),
                title=frappe._("Invalid TIN")
            )
        # Enforce single row per company
        existing = frappe.db.exists("AM Company Setup", {"company": self.company, "name": ["!=", self.name]})
        if existing:
            frappe.throw(
                frappe._("AM Company Setup already exists for {0}").format(self.company),
                frappe.DuplicateEntryError
            )

def validate_tin(tin: str) -> bool:
    """Armenian taxpayer identification number: exactly 8 digits."""
    if not tin or not isinstance(tin, str):
        return False
    return len(tin) == 8 and tin.isdigit()
```

## Pitfalls
- **`--custom` flag**: the DocType MUST be `custom = 1` so it lives in the database schema and survives ERPNext updates. Without this, a `bench update` will try to recreate it and fail.
- **Translation**: add a row to `apps/frappe_armenia/frappe_armenia/translations/hy.csv` for the DocType name and field labels. Example: `"AM Company Setup","Հայաստանի ընկերության կարգավորումներ"`.
- **Hook into Company**: in a later task (W1-T02), add `doc_events = {"Company": {"on_update": "frappe_armenia.regional.doctype.am_company_setup.am_company_setup.create_if_missing"}}`. NOT in this task.
