### Step 3 â€” Implement helpers
```bash
# infra/scripts/lib-upstream.sh
# Pure bash helpers for sync-upstream.sh. No network calls.

parse_minor() {
    # $1 = tag like "v15.3.0" or "v15.3.0-beta". Returns "15.3".
    local tag="$1"
    local stripped="${tag#v}"                              # "15.3.0" or "15.3.0-beta"
    local core="${stripped%%-*}"                           # "15.3.0"
    local major="${core%%.*}"                              # "15"
    local rest="${core#*.}"                                # "3.0"
    local minor="${rest%%.*}"                             # "3"
    echo "${major}.${minor}"
}

version_gt() {
    # $1 > $2 in dotted version? Both like "15.3.0".
    local highest=$(printf "%s\n%s\n" "$1" "$2" | sort -V | tail -1)
    [[ "$highest" == "$1" ]] && [[ "$1" != "$2" ]]
}

is_newer_than_pin() {
    # $1 = upstream tag (e.g. v15.4.0), $2 = pinned version (e.g. 15.3.0)
    local upstream="$1"
    local pin="$2"
    local u="${upstream#v}"
    version_gt "$u" "$pin"
}

within_compat_window() {
    # $1 = upstream tag, $2 = our current pin. True if upstream is within
    # 2 minor versions of our pin (so we ship against last 3 minors total).
    local upstream_minor=$(parse_minor "$1")
    local pin_minor="$2"
    local u_major="${upstream_minor%%.*}"
    local u_minor="${upstream_minor#*.}"
    local p_major="${pin_minor%%.*}"
    local p_minor="${pin_minor#*.}"
    if [[ "$u_major" != "$p_major" ]]; then
        [[ $((u_major - p_major)) -le 1 ]]   # allow one major jump (unusual)
        return
    fi
    local diff=$((p_minor - u_minor))
    [[ $diff -le 2 ]]
}
```

Rewrite `infra/scripts/sync-upstream.sh` `detect` to use them:
```bash
detect() {
    local changes=false
    declare -A seen_minor=()
    for entry in "${REPOS[@]}"; do
        local repo="${entry%%:*}"
        local current_pin="${entry##*:}"
        local latest
        latest=$(gh release list --repo "$repo" --limit 50 --json tagName -q \
                  '[.[] | select(.tagName | startswith("v15."))][0].tagName' 2>/dev/null || echo "")
        if [[ -z "$latest" ]]; then
            echo "WARN: no upstream tag for $repo" >&2
            continue
        fi
        if is_newer_than_pin "$latest" "$current_pin"; then
            echo "  $repo: latest=$latest is newer than pin=$current_pin"
            changes=true
        fi
    done
    if $changes; then
        [[ -n "${GITHUB_OUTPUT:-}" ]] && echo "changes=true" >> "$GITHUB_OUTPUT"
        echo "true"
    else
        [[ -n "${GITHUB_OUTPUT:-}" ]] && echo "changes=false" >> "$GITHUB_OUTPUT"
        echo "false"
    fi
}
```

## Pitfalls
- **BSD vs GNU `sort -V`**: macOS uses BSD sort which lacks `-V`. The detect function runs in CI (Linux) so this is fine for production, but the bash tests should be runnable on macOS. Replace `sort -V` with the loop trick if needed:
  ```bash
  version_gt() {
      local IFS=.
      local a=($1) b=($2)
      for i in 0 1 2; do
          if (( ${a[$i]:-0} > ${b[$i]:-0} )); then return 0; fi
          if (( ${a[$i]:-0} < ${b[$i]:-0} )); then return 1; fi
      done
      return 1  # equal
  }
  ```
- **`gh` rate limit**: `gh release list` can hit the GitHub API 60Ă—/h unauthenticated. Use `gh auth status` to ensure auth. The script will run in CI where the GITHUB_TOKEN provides 1000Ă—/h.
- **Pre-release tags**: we ignore them (no `v15.4.0-beta` should bump the pin).
- **Skip `frappe/hrms` if it's not tagged v15.x**: some HRMS releases live in older branches.
